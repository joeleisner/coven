> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ship the v0.2.0 API: charm/hex twin pairs, `$wake`/`$soul` lifecycle primitives, namespaced exports with subpath entries, a rebuilt playground, and a docs pass.

**Architecture:** New charm twins (`$shdw`, `$template`, `$on`, `$wake`) live under `src/charms/`. Updated and new hexes (`$shdw`, `$template`, `$on`, `$soul`) under `src/hexes/` call `$bewitch` on entry and use their charm twin for the core operation. The main entry (`src/mod.ts`) switches to namespaced `hexes`/`charms` exports; `deno.json` adds per-hex deep subpath entries so grimoire symbols are reachable for advanced users.

**Tech Stack:** Deno, TypeScript, happy-dom (tests), Vite (playground dev server), PicoCSS CDN (playground styles), JSR publish.

export function $shdw(
	component: HTMLElement,
	html?: HTMLTemplateElement | string,
): ShadowRoot {
	if (!component.shadowRoot) {
		component.attachShadow({ mode: 'open' });
		if (html) {
			let template: HTMLTemplateElement;
			if (typeof html === 'string') {
				template = document.createElement('template');
				template.innerHTML = html;
			} else {
				template = html;
			}
			component.shadowRoot!.appendChild(template.content.cloneNode(true));
		}
	}
	return component.shadowRoot!;
}

export default $shdw;
```

export function $wake(
	element: HTMLElement,
	callback: () => void,
	signal?: AbortSignal,
): void {
	if (document.readyState !== 'loading') {
		callback();
	} else {

	}
}

export default $wake;

import { $shdw as $shdwCharm } from '../charms/$shdw.ts';
import { $bewitch } from './$bewitch.ts';
import { $template } from './$template.ts';
import { $mut } from './$mut.ts';
import { grimoire, type GrimoireElement } from '../grimoire.ts';

/** @advanced Direct access to $shdw's grimoire slot. */
export const $SHDW_GRIMOIRE_SYMBOL = Symbol('$shdw');

type $ShdwGrimoire = {
	parts?: Set<string>;
};

function collectParts(root: ParentNode, into: Set<string>): void {
	for (const node of root.querySelectorAll('[part]')) {
		const raw = node.getAttribute('part') ?? '';
		for (const token of raw.split(/\s+/)) {
			if (token) into.add(token);
		}
	}
}

export function $shdw(
	component: HTMLElement,
	html?: HTMLTemplateElement | string,
): ShadowRoot {
	$bewitch(component);

	if (typeof html === 'string') html = $template(component, html);

	const store = grimoire<$ShdwGrimoire>(
		component as GrimoireElement,
		$SHDW_GRIMOIRE_SYMBOL,
	);
	store.parts ??= new Set();

	const firstCall = !component.shadowRoot;

	$shdwCharm(component, html);

	collectParts(component.shadowRoot!, store.parts);

	if (firstCall) {
		$mut(component.shadowRoot!, {
			type: 'childList',
			subtree: true,
			callback: () => {
				collectParts(component.shadowRoot!, store.parts!);
				propagate(component);
			},
		});

		queueMicrotask(() => propagate(component));
	}

	return component.shadowRoot!;
}

$shdw.parts = (element: HTMLElement): ReadonlySet<string> | undefined =>
	grimoire<$ShdwGrimoire>(
		element as GrimoireElement,
		$SHDW_GRIMOIRE_SYMBOL,
	).parts;

$shdw.root = (element: HTMLElement): ShadowRoot | null => element.shadowRoot;

$shdw.propagate = (element: HTMLElement): void => propagate(element);

function propagate(element: HTMLElement): void {
	const parts = grimoire<$ShdwGrimoire>(
		element as GrimoireElement,
		$SHDW_GRIMOIRE_SYMBOL,
	).parts;
	if (!parts || !parts.size) return;

	let node: Node | null = element.parentNode;
	while (node && !(node instanceof ShadowRoot)) {
		node = node.parentNode;
	}
	if (!node) return;

	const existing = (element.getAttribute('exportparts') ?? '')
		.split(',')
		.map((s) => s.trim())
		.filter(Boolean);
	const merged = new Set<string>([...existing, ...parts]);
	element.setAttribute('exportparts', [...merged].join(', '));
}

export default $shdw;
```

Key changes from the original:
- Added `import { $shdw as $shdwCharm } from '../charms/$shdw.ts'`
- Added `$bewitch(component)` as first line of `$shdw`
- Replaced the inline `attachShadow` + `appendChild` block with `$shdwCharm(component, html)` (after the `firstCall` flag is captured)
- Changed `const $SHDW_GRIMOIRE_SYMBOL` to `export const $SHDW_GRIMOIRE_SYMBOL`

/** @advanced Direct access to $template's grimoire slot. */
export const $TEMPLATE_GRIMOIRE_SYMBOL = Symbol('$template');

type $TemplateGrimoire = {
	templates?: Map<string, HTMLTemplateElement>;
};

export function $template(
	element: HTMLElement,
	html: string,

	$bewitch(element);

		element as GrimoireElement,

$template.clone = (

	html: string,
): DocumentFragment => $template(element, html).content.cloneNode(true) as DocumentFragment;

export default $template;

import { $bewitch } from './$bewitch.ts';
import { $wake } from '../charms/$wake.ts';
import { grimoire, type GrimoireElement } from '../grimoire.ts';

/** @advanced Direct access to $soul's grimoire slot. */
export const $SOUL_GRIMOIRE_SYMBOL = Symbol('$soul');

type $SoulGrimoire = {
	bound?: boolean;
};

export type $SoulOptions = {
	connected: (signal: AbortSignal) => void;
	disconnected?: () => void;
};

export function $soul(
	element: HTMLElement,
	{ connected, disconnected }: $SoulOptions,
): void {
	const existing = $bewitch.signal(element);
	const signal = existing?.aborted ? $bewitch.renew(element) : $bewitch(element);

	const store = grimoire<$SoulGrimoire>(
		element as GrimoireElement,
		$SOUL_GRIMOIRE_SYMBOL,
	);
	store.bound = true;

	$wake(element, () => {
		connected(signal);
		if (disconnected) {
			signal.addEventListener('abort', disconnected, { once: true });
		}
	}, signal);
}

export default $soul;

The log wraps an `<ol>` in light DOM. `$attr` reads the `listen` attribute; `$on` registers per-type listeners on `document`. Each event's `detail` is the message. A private `AbortController` per listener batch allows re-registration when the attribute changes without aborting the element's primary signal.

```ts
import { Familiar, hexes, charms } from '../mod.ts';

charms.$define(
	'coven-log',
	class CovenLog extends Familiar {
		#listenerCtrl: AbortController | null = null;

		override connected(): void {
			hexes.$attr<string>(this, {
				name: 'listen',
				value: '',
				callback: (value) => this.#rebind(value),
			});
		}

		override disconnected(): void {
			this.#listenerCtrl?.abort();
		}

		#rebind(types: string): void {
			this.#listenerCtrl?.abort();
			this.#listenerCtrl = new AbortController();
			const { signal } = this.#listenerCtrl;

			for (const type of types.trim().split(/\s+/).filter(Boolean)) {
				document.addEventListener(
					type,
					(event) => {
						if (!(event instanceof CustomEvent)) return;
						const li = document.createElement('li');
						li.textContent = String(event.detail ?? type);
						this.querySelector('ol')?.prepend(li);
					},
					{ signal },
				);
			}
		}
	},
);
```

import { Familiar, hexes, charms } from '../mod.ts';

charms.$define(
	'coven-toggle',
	class CovenToggle extends Familiar {
		declare open: boolean;

		override setup(): void {
			hexes.$shdw(
				this,
				/*html*/ `
				<button part="trigger">Hide</button>
				<slot></slot>
			`,
			);
		}

		override connected(): void {
			hexes.$prop<boolean>(this, { name: 'open', value: true });

			const trigger = this.shadowRoot!.querySelector<HTMLElement>('[part="trigger"]')!;
			const slot = this.shadowRoot!.querySelector<HTMLElement>('slot')!;

			charms.$on(trigger, {
				type: 'click',
				signal: this.signal,
				callback: () => {
					this.open = !this.open;
					slot.style.display = this.open ? '' : 'none';
					trigger.textContent = this.open ? 'Hide' : 'Show';
					charms.$emit(this, { detail: this.open ? 'shown' : 'hidden' });
				},
			});

			hexes.$scry(this, {
				callback: (entries) => {
					const intersecting = entries.some((e) => e.isIntersecting);
					charms.$emit(this, {
						detail: intersecting ? 'entered viewport' : 'exited viewport',
					});
				},
			});
		}
	},
);

import { $bewitch } from './$bewitch.ts';
import { $wake } from '../charms/$wake.ts';

export function $soul(
	element: HTMLElement,
	{ connected, disconnected }: { connected: (signal: AbortSignal) => void; disconnected?: () => void },
): void {
	$bewitch(element); // renew if aborted for reconnection
	const signal = $bewitch.signal(element)!;

	$wake(element, () => {
		connected(signal);
		if (disconnected) {
			signal.addEventListener('abort', disconnected, { once: true });
		}
	}, signal);
}
